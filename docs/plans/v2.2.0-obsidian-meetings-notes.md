import { homedir } from 'os';
import { existsSync, readdirSync, statSync, readFileSync } from 'fs';
import { join, basename } from 'path';

const DEFAULT_VAULT_PATHS = [
  join(homedir(), 'Documents', 'Obsidian Vault'),
  join(homedir(), 'Obsidian Vault'),
  join(process.cwd(), 'data', 'obsidian-vault')
];

export function detectVault() {
  for (const p of DEFAULT_VAULT_PATHS) {
    if (existsSync(p)) return { path: p, exists: true };
  }
  const fallback = join(process.cwd(), 'data', 'obsidian-vault');
  return { path: fallback, exists: false };
}

export function listFolders(vaultPath = detectVault().path) {
  if (!existsSync(vaultPath)) return [];
  return readdirSync(vaultPath)
    .filter(f => statSync(join(vaultPath, f)).isDirectory() &amp;&amp; !f.startsWith('.'))
    .map(f => ({ name: f, path: join(vaultPath, f) }));
}

export function listNotes(folderPath) {
  if (!existsSync(folderPath)) return [];
  return readdirSync(folderPath)
    .filter(f => f.endsWith('.md'))
    .map(f => ({ name: f, path: join(folderPath, f) }));
}

export function readNote(notePath) {
  if (!existsSync(notePath)) return null;
  return readFileSync(notePath, 'utf8');
}
```

export function ticketToNote(ticket) {
  return [
    `# Ticket ${ticket.id}`,
    '',
    `- **Subject:** ${ticket.subject}`,
    `- **Status:** ${ticket.status}`,
    `- **Priority:** ${ticket.priority}`,
    `- **Created:** ${ticket.created_at}`,
    '',
    '## Body',
    ticket.body || '',
    '',
    '## Actionables',
    '- [ ] Verificar con usuario',
    '- [ ] Escalar si persiste',
    ''
  ].join('\n');
}

export function campaignToNote(campaign) {
  return [
    `# Campaign: ${campaign.title || campaign.id}`,
    '',
    `- **Status:** ${campaign.status || 'draft'}`,
    `- **Agents:** ${(campaign.agents || []).join(', ')}`,
    '',
    '## Outputs',
    '```',
    JSON.stringify(campaign.results || {}, null, 2),
    '```',
    ''
  ].join('\n');
}

export function analyticsToNote(analytics) {
  return [
    '# Analytics Summary',
    '',
    `- **Period:** ${analytics.period}`,
    `- **Total Calls:** ${analytics.total_calls}`,
    `- **Total Cost:** $${(analytics.total_cost || 0).toFixed(4)}`,
    '',
    '## Top Models',
    analytics.top_models ? analytics.top_models.map(m => `- ${m.model}: ${m.calls} calls`).join('\n') : '',
    ''
  ].join('\n');
}
```

import helpdeskService from '../helpdesk/helpdeskService.js';
import * as aiNotesService from './aiNotesService.js';

export function processSummary({ title, summary, tenant_id, date = new Date().toISOString() }) {
  const sentences = summary.split(/[.\n]/).map(s => s.trim()).filter(Boolean);
  const actionItems = [];
  for (const s of sentences) {
    if (/\b(revisar|actualizar|implementar|corregir|probar|desplegar|investigar|enviar|revisiĂłn)\b/i.test(s)) {
      let ticketId = null;
      const match = s.match(/\b(ticket|issue|TK|INC)[-\s]?(\w+)/i);
      if (match) {
        ticketId = match[0];
      }
      actionItems.push({ text: s, owner: extractOwner(s), ticket_id: ticketId });
    }
  }

  // Optionally create follow-up ticket for non-linked actions
  const orphanActions = actionItems.filter(a => !a.ticket_id);
  const createdTickets = orphanActions.slice(0, 3).map(a =>
    helpdeskService.createTicket({
      tenant_id,
      requester_email: 'meeting@azurdesk.ai',
      requester_name: 'Teams Pipeline',
      subject: `AcciĂłn reuniĂłn: ${a.text.slice(0, 60)}`,
      body: `Generado desde resumen de Teams:\n${summary}\n\nAcciĂłn: ${a.text}\nOwner: ${a.owner || 'unassigned'}`
    })
  );

  const note = aiNotesService.campaignToNote({
    id: `meeting-${date.slice(0, 10)}`,
    title,
    status: 'processed',
    results: { summary, action_items: actionItems, tickets_created: createdTickets.length }
  });

  return { title, action_items: actionItems, tickets_created: createdTickets, note };
}

function extractOwner(text) {
  const m = text.match(/\b(Juan|MarĂ­a|Pedro|Ana|Luis|Carlos|SofĂ­a|[A-Z][a-z]+)\b/);
  return m ? m[1] : null;
}
```
